\begin{frame}{Why MAB, CMAB, and iCMAB}
\small
\begin{tabular}{p{0.10\textwidth}p{0.32\textwidth}p{0.50\textwidth}}
\toprule
\textbf{Policy level} & \textbf{What it sees} & \textbf{What it tests} \\
\midrule

\rowcolor{phase1blue!12}
\mab &
\textbf{RQ1:} \textit{What happens without context?}
\newline
{\scriptsize (action history only)} &
\textbf{Phase 1 baseline:} establishes utility and fairness when the policy ignores context, giving the non-contextual comparison point for both domains. \\

\rowcolor{phase2green!12}
\cmab &
\textbf{RQ2:}\textit{ What happens when observed context is incomplete or degraded?}
\newline
{\scriptsize (observed context)} &
\textbf{Phase 2 context test:} measures whether observed context improves utility/fairness and how quickly gaps appear when context is missing, noisy, delayed, or group-unequal. \\

\rowcolor{phase3orange!12}
\icmab &
\textbf{RQ3--RQ4:} \textit{Can informative or recovered context reduce harm and transfer across domains?}
\newline
{\scriptsize (observed context plus informative or recovered context)} &
\textbf{Phase 3--4 mitigation test:} evaluates whether missingness-aware or fairness-aware updates reduce disparity without unacceptable utility loss, and whether the same logic transfers across clinical and quantum routing. \\

\bottomrule
\end{tabular}

% \vspace{0.7em}
\begin{block}{Design rationale}
The policy spectrum directly supports the research questions: \mab{} gives the no-context baseline, \cmab{} tests the value and failure modes of observed context, and \icmab{} tests whether recovered or fairness-aware context can reduce routing harm across both domains.
\end{block}
\end{frame}
